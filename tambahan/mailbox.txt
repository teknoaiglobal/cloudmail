    <script>
        // --- CONSTANTS & STATE ---
        const API_BASE = "https://api.mail.tm";
        
        let state = {
            account: null,
            token: null,
            messages: [],
            domains: [], // Added domains list
            savedAccounts: [], // New: List of saved accounts
            selectedMessage: null,
            isLoading: false,
            isRefreshing: false,
            error: null,
            autoRefresh: true,
            refreshInterval: null,
            authMode: 'create' // 'create' | 'login'
        };

        // --- HELPER FUNCTIONS ---
        const generateRandomString = (length) => {
            const chars = 'abcdefghijklmnopqrstuvwxyz0123456789';
            let result = '';
            for (let i = 0; i < length; i++) {
                result += chars.charAt(Math.floor(Math.random() * chars.length));
            }
            return result;
        };

        const copyToClipboard = (text) => {
            const textArea = document.createElement("textarea");
            textArea.value = text;
            document.body.appendChild(textArea);
            textArea.select();
            document.execCommand('copy');
            document.body.removeChild(textArea);
            alert("Disalin ke clipboard: " + text);
        };

        // Fill inputs with random data for quick creation
        const fillRandomData = () => {
            const userInput = document.getElementById('reg-username');
            const passInput = document.getElementById('reg-password');
            if (userInput && passInput) {
                userInput.value = generateRandomString(10);
                passInput.value = generateRandomString(12);
            }
        };

        // --- ACCOUNT VAULT FUNCTIONS (New) ---

        function toggleSaveAccount() {
            if (!state.account) return;

            const index = state.savedAccounts.findIndex(a => a.address === state.account.address);
            let newSaved = [...state.savedAccounts];

            if (index >= 0) {
                // Remove
                newSaved.splice(index, 1);
            } else {
                // Add
                newSaved.push(state.account);
            }

            state.savedAccounts = newSaved;
            localStorage.setItem('tm_saved_accounts', JSON.stringify(newSaved));
            renderAccountSection();
        }

        function deleteSavedAccount(address, event) {
            if(event) event.stopPropagation();
            const newSaved = state.savedAccounts.filter(a => a.address !== address);
            state.savedAccounts = newSaved;
            localStorage.setItem('tm_saved_accounts', JSON.stringify(newSaved));
            renderAccountSection();
        }

        function loginSavedAccount(address) {
            const acc = state.savedAccounts.find(a => a.address === address);
            if (acc) {
                executeLogin(acc.address, acc.password);
            }
        }

        // --- API FUNCTIONS ---

        async function fetchDomains() {
            try {
                const res = await fetch(`${API_BASE}/domains`);
                if (res.ok) {
                    const data = await res.json();
                    updateState({ domains: data['hydra:member'] || [] });
                    // If we are on create screen, re-render to populate dropdown
                    if (state.authMode === 'create' && !state.account) {
                        renderAccountSection();
                    }
                }
            } catch (e) {
                console.error("Gagal mengambil domain", e);
            }
        }

        function toggleAuthMode() {
            updateState({ 
                authMode: state.authMode === 'create' ? 'login' : 'create',
                error: null 
            });
            render();
            
            // If switching to create mode, ensure domains are loaded and try to pre-fill random
            if (state.authMode === 'create') {
                if (state.domains.length === 0) fetchDomains();
                // Delay slightly to ensure DOM is ready
                setTimeout(fillRandomData, 100);
            }
        }

        async function loginAccount() {
            const emailInput = document.getElementById('login-email');
            const passwordInput = document.getElementById('login-password');
            const address = emailInput ? emailInput.value.trim() : '';
            const password = passwordInput ? passwordInput.value.trim() : '';

            if (!address || !password) {
                updateState({ error: "Email dan password harus diisi." });
                renderAccountSection(); // Re-render to show error
                return;
            }

            executeLogin(address, password);
        }

        async function createCustomAccount() {
            const usernameInput = document.getElementById('reg-username');
            const domainSelect = document.getElementById('reg-domain');
            const passwordInput = document.getElementById('reg-password');

            const username = usernameInput ? usernameInput.value.trim() : '';
            const domain = domainSelect ? domainSelect.value : '';
            const password = passwordInput ? passwordInput.value.trim() : '';

            if (!username || !domain || !password) {
                updateState({ error: "Mohon lengkapi username, domain, dan password." });
                renderAccountSection();
                return;
            }

            const address = `${username}@${domain}`;

            updateState({ isLoading: true, error: null });
            renderAccountSection();

            try {
                // 1. Create Account
                const accountRes = await fetch(`${API_BASE}/accounts`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ address, password })
                });

                if (!accountRes.ok) {
                    const errData = await accountRes.json();
                    throw new Error(errData['hydra:description'] || "Gagal membuat akun. Username mungkin sudah dipakai.");
                }

                const accountData = await accountRes.json();

                // 2. Login directly
                await executeLogin(address, password);

            } catch (err) {
                updateState({ error: err.message, isLoading: false });
                renderAccountSection();
            }
        }

        async function executeLogin(address, password) {
            updateState({ isLoading: true, error: null });
            renderAccountSection();

            try {
                // Get Token
                const tokenRes = await fetch(`${API_BASE}/token`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ address, password })
                });

                if (!tokenRes.ok) throw new Error("Autentikasi gagal. Cek kembali email/password.");
                const tokenData = await tokenRes.json();

                // Get Account Details
                const meRes = await fetch(`${API_BASE}/me`, {
                    headers: { 'Authorization': `Bearer ${tokenData.token}` }
                });

                if (!meRes.ok) throw new Error("Gagal mengambil data akun.");
                const accountData = await meRes.json();

                const newAccount = { ...accountData, password };
                
                localStorage.setItem('tm_account', JSON.stringify(newAccount));
                localStorage.setItem('tm_token', tokenData.token);

                // Check if this account is already saved/locked, if so update the object
                const savedIdx = state.savedAccounts.findIndex(a => a.address === newAccount.address);
                if (savedIdx >= 0) {
                    const newSaved = [...state.savedAccounts];
                    newSaved[savedIdx] = newAccount;
                    state.savedAccounts = newSaved;
                    localStorage.setItem('tm_saved_accounts', JSON.stringify(newSaved));
                }

                updateState({ 
                    account: newAccount, 
                    token: tokenData.token,
                    isLoading: false 
                });
                
                startAutoRefresh();
                render(); // Full render on success

            } catch (err) {
                updateState({ error: err.message, isLoading: false });
                renderAccountSection();
            }
        }

        async function fetchMessages(isAuto = false) {
            if (!state.token) return;
            
            if (!isAuto) {
                updateState({ isRefreshing: true });
                renderRefreshButton();
            }

            try {
                const res = await fetch(`${API_BASE}/messages`, {
                    headers: { 'Authorization': `Bearer ${state.token}` }
                });
                
                if (res.status === 401) {
                    logout();
                    return;
                }

                const data = await res.json();
                updateState({ messages: data['hydra:member'] || [] });
                
                if (!isAuto) updateState({ isRefreshing: false });
                
                renderInbox();
                renderRefreshButton();

            } catch (err) {
                console.error("Error fetching messages:", err);
                if (!isAuto) {
                    updateState({ isRefreshing: false });
                    renderRefreshButton();
                }
            }
        }

        async function readMessage(msgId) {
            if (!state.token) return;
            
            const tempSelected = state.messages.find(m => m.id === msgId);
            updateState({ selectedMessage: tempSelected });
            renderInbox();
            renderMessageView(true);

            try {
                const res = await fetch(`${API_BASE}/messages/${msgId}`, {
                    headers: { 'Authorization': `Bearer ${state.token}` }
                });
                const data = await res.json();
                
                const updatedMessages = state.messages.map(m => 
                    m.id === msgId ? { ...m, seen: true } : m
                );

                updateState({ 
                    selectedMessage: data,
                    messages: updatedMessages
                });

                renderInbox();
                renderMessageView(false);

            } catch (err) {
                console.error("Gagal membuka pesan", err);
                alert("Gagal membuka pesan.");
            }
        }

        function logout() {
            localStorage.removeItem('tm_account');
            localStorage.removeItem('tm_token');
            stopAutoRefresh();
            updateState({
                account: null,
                token: null,
                messages: [],
                selectedMessage: null
            });
            render();
            
            // After logout, try to fetch domains again to ensure dropdown is ready if they want to register again
            fetchDomains();
            setTimeout(fillRandomData, 100);
        }

        // --- STATE MANAGEMENT HELPERS ---
        
        function updateState(newState) {
            state = { ...state, ...newState };
        }

        function startAutoRefresh() {
            if (state.refreshInterval) clearInterval(state.refreshInterval);
            if (state.autoRefresh && state.token) {
                fetchMessages(true);
                state.refreshInterval = setInterval(() => fetchMessages(true), 10000);
            }
        }

        function stopAutoRefresh() {
            if (state.refreshInterval) clearInterval(state.refreshInterval);
        }

        function toggleAutoRefresh() {
            updateState({ autoRefresh: !state.autoRefresh });
            if (state.autoRefresh) {
                startAutoRefresh();
            } else {
                stopAutoRefresh();
            }
            renderHeaderControls();
        }

        // --- RENDER FUNCTIONS (UI) ---

        function render() {
            renderAccountSection();
            renderInbox();
            renderMessageView();
            renderHeaderControls();
            lucide.createIcons();
        }

        function renderHeaderControls() {
            const btnAuto = document.getElementById('btn-autorefresh');
            const isActive = state.autoRefresh;
            
            btnAuto.className = `p-1.5 rounded-lg transition-all border ${isActive ? 'text-green-400 bg-green-900/30 border-green-500/30' : 'text-slate-500 border-transparent hover:bg-white/5'}`;
            btnAuto.innerHTML = `<div class="w-2 h-2 rounded-full ${isActive ? 'bg-green-400 animate-pulse shadow-[0_0_8px_rgba(74,222,128,0.8)]' : 'bg-slate-600'}"></div>`;
            
            // Message Count
            const countBadge = document.getElementById('msg-count');
            if(state.messages.length > 0) {
                countBadge.classList.remove('hidden');
                countBadge.innerText = state.messages.length;
            } else {
                countBadge.classList.add('hidden');
            }
        }

        function renderRefreshButton() {
            const btn = document.getElementById('btn-refresh');
            const iconClass = state.isRefreshing ? 'animate-spin' : '';
            btn.innerHTML = `<i data-lucide="refresh-cw" class="w-4 h-4 ${iconClass}"></i>`;
            lucide.createIcons();
        }

        function renderAccountSection() {
            const container = document.getElementById('account-section');
            
            // Render Saved Accounts List (Shared Component)
            // Helper untuk list akun yang disimpan (sekarang dijadikan string HTML langsung di bawah agar lebih fleksibel)
            
            if (!state.account) {
                // Not Logged In View
                if (state.authMode === 'login') {
                    // --- LOGIN FORM ---
                    container.innerHTML = `
                        <div class="text-center py-4 animate-in fade-in slide-in-from-left-4 duration-300">
                            <h3 class="text-cyan-100 text-sm font-bold mb-4 uppercase tracking-wider">Login Akun</h3>
                            
                            <div class="space-y-3 mb-4">
                                <input type="email" id="login-email" placeholder="Email Address" 
                                    class="w-full bg-white/5 border border-white/10 rounded-lg px-3 py-2 text-sm text-white focus:outline-none focus:border-cyan-500/50 focus:bg-white/10 transition-all placeholder:text-slate-500 font-mono">
                                <input type="password" id="login-password" placeholder="Password" 
                                    class="w-full bg-white/5 border border-white/10 rounded-lg px-3 py-2 text-sm text-white focus:outline-none focus:border-cyan-500/50 focus:bg-white/10 transition-all placeholder:text-slate-500 font-mono">
                            </div>

                            <button onclick="loginAccount()" ${state.isLoading ? 'disabled' : ''}
                                class="w-full py-2.5 px-4 bg-gradient-to-r from-cyan-600 to-blue-600 hover:from-cyan-500 hover:to-blue-500 text-white rounded-xl font-bold tracking-wide transition-all flex items-center justify-center gap-2 shadow-[0_0_15px_rgba(6,182,212,0.3)] border border-cyan-400/30 ${state.isLoading ? 'opacity-70 cursor-not-allowed' : ''}">
                                ${state.isLoading ? '<i data-lucide="loader-2" class="w-4 h-4 animate-spin"></i>' : '<i data-lucide="log-in" class="w-4 h-4"></i>'}
                                MASUK
                            </button>

                            <button onclick="toggleAuthMode()" class="mt-4 text-xs text-slate-400 hover:text-cyan-300 underline underline-offset-2 transition-colors">
                                Belum punya akun? Buat baru
                            </button>

                            ${state.error ? `
                            <div class="mt-4 p-3 bg-red-500/20 border border-red-500/30 backdrop-blur-md rounded-lg flex items-start gap-2 text-left">
                                <i data-lucide="alert-circle" class="w-4 h-4 text-red-400 mt-0.5 shrink-0"></i>
                                <p class="text-xs text-red-200">${state.error}</p>
                            </div>` : ''}

                            <!-- Simple list for login screen -->
                            ${state.savedAccounts.length > 0 ? `
                                <div class="mt-6 pt-4 border-t border-white/10">
                                    <h4 class="text-[10px] font-bold text-slate-500 uppercase tracking-widest mb-3 flex items-center justify-center gap-2">
                                        <i data-lucide="history" class="w-3 h-3"></i> Riwayat
                                    </h4>
                                    <div class="space-y-1">
                                        ${state.savedAccounts.map(acc => `
                                            <button onclick="loginSavedAccount('${acc.address}')" class="w-full flex items-center justify-between p-2 rounded hover:bg-white/5 text-xs text-slate-400 hover:text-cyan-300 transition-colors font-mono">
                                                <span class="truncate">${acc.address}</span>
                                                <i data-lucide="chevron-right" class="w-3 h-3 opacity-50"></i>
                                            </button>
                                        `).join('')}
                                    </div>
                                </div>
                            ` : ''}
                        </div>
                    `;
                } else {
                    // --- CREATE FORM (Custom & Random) ---
                    const domainOptions = state.domains.length > 0 
                        ? state.domains.map(d => `<option value="${d.domain}">@${d.domain}</option>`).join('')
                        : '<option disabled>Loading domains...</option>';

                    container.innerHTML = `
                        <div class="text-center py-4 animate-in fade-in slide-in-from-right-4 duration-300">
                            <div class="flex items-center justify-between mb-4">
                                <h3 class="text-cyan-100 text-sm font-bold uppercase tracking-wider">Buat Akun Baru</h3>
                                <button onclick="fillRandomData()" title="Isi data acak" class="p-1.5 bg-white/5 hover:bg-white/10 rounded-md text-cyan-400 transition-colors">
                                    <i data-lucide="dices" class="w-4 h-4"></i>
                                </button>
                            </div>

                            <div class="space-y-3 mb-4">
                                <div class="flex gap-2">
                                    <input type="text" id="reg-username" placeholder="User" 
                                        class="w-full min-w-0 bg-white/5 border border-white/10 rounded-lg px-3 py-2 text-sm text-white focus:outline-none focus:border-cyan-500/50 focus:bg-white/10 transition-all placeholder:text-slate-500 font-mono">
                                </div>
                                
                                <select id="reg-domain" class="w-full bg-slate-800 border border-white/10 rounded-lg px-3 py-2 text-sm text-slate-300 focus:outline-none focus:border-cyan-500/50 transition-all font-mono appearance-none">
                                    ${domainOptions}
                                </select>
                                
                                <input type="password" id="reg-password" placeholder="Password" 
                                    class="w-full bg-white/5 border border-white/10 rounded-lg px-3 py-2 text-sm text-white focus:outline-none focus:border-cyan-500/50 focus:bg-white/10 transition-all placeholder:text-slate-500 font-mono">
                            </div>

                            <button onclick="createCustomAccount()" ${state.isLoading ? 'disabled' : ''}
                                class="w-full py-2.5 px-4 bg-gradient-to-r from-cyan-600 to-blue-600 hover:from-cyan-500 hover:to-blue-500 text-white rounded-xl font-bold tracking-wide transition-all flex items-center justify-center gap-2 shadow-[0_0_20px_rgba(6,182,212,0.5)] border border-cyan-400/30 ${state.isLoading ? 'opacity-70 cursor-not-allowed' : ''}">
                                ${state.isLoading ? '<i data-lucide="loader-2" class="w-4 h-4 animate-spin"></i>' : '<i data-lucide="zap" class="w-4 h-4 fill-current"></i>'}
                                BUAT AKUN
                            </button>
                            
                            <button onclick="toggleAuthMode()" class="mt-4 text-xs text-slate-400 hover:text-cyan-300 underline underline-offset-2 transition-colors">
                                Sudah punya akun? Login manual
                            </button>

                            ${state.error ? `
                            <div class="mt-4 p-3 bg-red-500/20 border border-red-500/30 backdrop-blur-md rounded-lg flex items-start gap-2 text-left">
                                <i data-lucide="alert-circle" class="w-4 h-4 text-red-400 mt-0.5 shrink-0"></i>
                                <p class="text-xs text-red-200">${state.error}</p>
                            </div>` : ''}
                        </div>
                    `;
                }
            } else {
                // --- LOGGED IN VIEW (MINIMALIST) ---
                const isSaved = state.savedAccounts.some(a => a.address === state.account.address);
                const savedCount = state.savedAccounts.length;
                const avatarChar = state.account.address[0].toUpperCase();
                
                container.innerHTML = `
                    <div class="animate-in fade-in slide-in-from-left-4 duration-500">
                        
                        <!-- Main Account Card -->
                        <div class="bg-gradient-to-br from-white/10 to-white/5 rounded-xl p-3 border border-white/10 shadow-lg relative group">
                            
                            <!-- Header: Avatar & Email -->
                            <div class="flex items-center gap-3 mb-3">
                                <div class="w-10 h-10 shrink-0 rounded-lg bg-gradient-to-br from-cyan-500 to-blue-600 flex items-center justify-center text-white font-bold text-lg shadow-inner border border-white/20">
                                    ${avatarChar}
                                </div>
                                <div class="min-w-0 flex-1">
                                    <div class="flex items-center justify-between">
                                        <h3 class="text-[10px] font-bold text-slate-400 uppercase tracking-wider">Active ID</h3>
                                        <button onclick="copyToClipboard('${state.account.address}')" class="text-slate-500 hover:text-cyan-300 transition-colors" title="Copy Email">
                                            <i data-lucide="copy" class="w-3 h-3"></i>
                                        </button>
                                    </div>
                                    <p class="text-sm font-mono text-cyan-50 truncate" title="${state.account.address}">${state.account.address}</p>
                                </div>
                            </div>

                            <!-- Action Row (Minimalist Buttons) -->

        // --- PARTICLE SYSTEM (Canvas) ---
        function initParticles() {
            const canvas = document.getElementById('particle-canvas');
            const ctx = canvas.getContext('2d');
            let particles = [];

            const resizeCanvas = () => {
                canvas.width = window.innerWidth;
                canvas.height = window.innerHeight;
            };

            const createParticles = () => {
                const particleCount = Math.min(window.innerWidth / 10, 100);
                particles = [];
                for (let i = 0; i < particleCount; i++) {
                    particles.push({
                        x: Math.random() * canvas.width,
                        y: Math.random() * canvas.height,
                        vx: (Math.random() - 0.5) * 1.5,
                        vy: (Math.random() - 0.5) * 1.5,
                        size: Math.random() * 2 + 1
                    });
                }
            };

            const drawParticles = () => {
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                
                particles.forEach((p, index) => {
                    p.x += p.vx;
                    p.y += p.vy;

                    if (p.x < 0 || p.x > canvas.width) p.vx *= -1;
                    if (p.y < 0 || p.y > canvas.height) p.vy *= -1;

                    ctx.beginPath();
                    ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
                    ctx.fillStyle = 'rgba(100, 200, 255, 0.7)';
                    ctx.fill();

                    for (let j = index + 1; j < particles.length; j++) {
                        const p2 = particles[j];
                        const dx = p.x - p2.x;
                        const dy = p.y - p2.y;
                        const distance = Math.sqrt(dx * dx + dy * dy);

                        if (distance < 150) {
                            ctx.beginPath();
                            ctx.strokeStyle = `rgba(100, 200, 255, ${1 - distance / 150})`;
                            ctx.lineWidth = 0.5;
                            ctx.moveTo(p.x, p.y);
                            ctx.lineTo(p2.x, p2.y);
                            ctx.stroke();
                        }
                    }
                });
                requestAnimationFrame(drawParticles);
            };

            window.addEventListener('resize', () => {
                resizeCanvas();
                createParticles();
            });

            resizeCanvas();
            createParticles();
            drawParticles();
        }

        // --- INIT APP ---
        window.addEventListener('DOMContentLoaded', () => {
